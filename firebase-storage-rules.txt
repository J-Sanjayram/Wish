// Firebase Storage Security Rules
// Add these rules in Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Only allow access to images with specific naming pattern
    match /birthday-wishes/{wishId}/{allPaths=**} {
      // Only allow read if user knows the exact wishId
      allow read: if request.auth != null || 
                     resource.name.matches('.*\\.(jpg|jpeg|png|gif|webp)$');
      
      // Only allow write for authenticated users
      allow write: if request.auth != null &&
                      request.resource.size < 5 * 1024 * 1024 && // 5MB limit
                      request.resource.contentType.matches('image/.*');
    }
    
    // Deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
