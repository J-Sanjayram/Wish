// Firebase Storage Security Rules - No Authentication Required
// Auto-delete after 24 hours

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow public read/write for birthday wishes images
    match /birthday-wishes/{wishId}/{allPaths=**} {
      // Allow read for anyone
      allow read: if true;
      
      // Allow write with size limit and image types only
      allow write: if request.resource.size < 5 * 1024 * 1024 && // 5MB limit
                      request.resource.contentType.matches('image/.*');
      
      // Allow delete (for cleanup)
      allow delete: if true;
    }
    
    // Deny all other paths
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
